from onnxruntime.quantization import quantize_dynamic, QuantType

FP32_MODEL = r"C:\Users\charlie.marzan\Downloads\Charlie Files\MIRI AI Project\Model EfficientNetB7 with ROI\ROI-det.onnx"
INT8_MODEL = r"C:\Users\charlie.marzan\Downloads\Charlie Files\MIRI AI Project\Model EfficientNetB7 with ROI\ROI-det-int8-linear.onnx"

# Quantize ONLY MatMul / Gemm (linear layers), skip Conv
quantize_dynamic(
    model_input=FP32_MODEL,
    model_output=INT8_MODEL,
    weight_type=QuantType.QInt8,
    op_types_to_quantize=["MatMul", "Gemm"],  # 👈 key line
)

print("Saved partially quantized model:", INT8_MODEL)
